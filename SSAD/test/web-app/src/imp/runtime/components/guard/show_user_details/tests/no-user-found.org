** Test Case

   Imports the TestHarness which sets up the component.  This test creates an
   instruction with cmd =ShowDetails=,a dictionary
   with an object that is an instance of =Email= and an existing session.  The guard's do
   method fails.

#+BEGIN_SRC python :tangle no_user_for_email.py
class TestDoPasses(TestHarness):

    def test_do_passes(self):
        print "test_do_passes"
        session = self.session
        email = Email(val = "nouser@gnu.org")
        instr = {'cmd': Cmd.show_user_details, 'session': session, 'data': {'email' : email}}
        guard = self.component
        result = guard.do(instr)
        print result
        self.assertEqual(result, instr)
