    @staticmethod 
    def do(obj, instr):
        print "set_name: %s" % instr
        user = instr['data']['user']
        setname = instr['data']['setname']
        setname = setname.get("val")
        obj.em.delete_user(user)
        user.set( name = Name(val=setname))
